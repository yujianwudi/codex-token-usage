/*

#include <dlfcn.h>

import "C"

import (
	"fmt"
	"unsafe"
)

type compatABIPanicControl struct {
	handle unsafe.Pointer
	setter unsafe.Pointer
	getter unsafe.Pointer
}

func openCompatABIPanicControl(path string) (*compatABIPanicControl, error) {
	cPath := C.CString(path)
	defer C.free(unsafe.Pointer(cPath))
	handle := C.ctu_panic_dlopen(cPath)
	if handle == nil {
		return nil, fmt.Errorf("dlopen panic harness: %s", compatABIPanicDLError())
	}
	resolve := func(name string) unsafe.Pointer {
		cName := C.CString(name)
		defer C.free(unsafe.Pointer(cName))
		return C.ctu_panic_dlsym(handle, cName)
	}
	setter := resolve("cliproxyTestSetPanicPoint")
	getter := resolve("cliproxyTestGetPanicPoint")
	if setter == nil || getter == nil {
		_ = C.ctu_panic_dlclose(handle)
		return nil, fmt.Errorf("panic harness control symbols are unavailable: %s", compatABIPanicDLError())
	}
	return &compatABIPanicControl{handle: handle, setter: setter, getter: getter}, nil
}

func (c *compatABIPanicControl) Set(point uint32) error {
	if c == nil || c.handle == nil || c.setter == nil {
		return fmt.Errorf("panic harness control is closed")
	}
	C.ctu_set_panic_point(c.setter, C.uint32_t(point))
	return nil
}

func (c *compatABIPanicControl) Current() uint32 {
	if c == nil || c.handle == nil || c.getter == nil {
		return 0
	}
	return uint32(C.ctu_get_panic_point(c.getter))
}

func (c *compatABIPanicControl) Close() {
	if c == nil || c.handle == nil {
		return
	}
	_ = C.ctu_panic_dlclose(c.handle)
	c.handle = nil
	c.setter = nil
	c.getter = nil
}

func compatABIPanicDLError() string {
	value := C.ctu_panic_dlerror()
	if value == nil {
		return "unknown dynamic-loader error"
	}
	return C.GoString(value)
}
